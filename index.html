Example: Random Intro Image

front-page.php (jQuery)

<script type="text/javascript">
    /* Display random intro image on front page */
    var introPhotoNum = Math.floor((Math.random()*4)+1); // Random number between 1 and 4

    if (jQuery('html').hasClass('ie-lt9')) {
        jQuery('.ie-lt9 #intro').css('background-image', "url('" + wp_urls.stylesheet_directory_uri + '/_/img/intro/ie/' + introPhotoNum + ".jpg')");
    } else {
        jQuery('#intro').css('background-image', "url('" + wp_urls.stylesheet_directory_uri + '/_/img/intro/' + introPhotoNum + ".jpg')");
    }
</script>
